## Overview

Phase 2 creates `src/expander.js` — a new module that transforms reader AST into compiler-ready AST by resolving quasiquote, quote, sugar forms (`cons`, `list`, `car`/`cdr`/`cadr`/`cddr`), and `as` patterns. The expander does NOT import from the compiler. It produces the same AST node types the compiler already understands.

**Pipeline change:**
```
.lykn source → reader → EXPANDER → compiler → astring → JS
```

---

## Module Structure

**Single file: `src/expander.js`** (~400–500 lines)

All functions closure-scoped or module-level. Gensym counter is module-level state.

**Exports:**
- `expand(forms)` — main entry, takes `AstNode[]`, returns `AstNode[]`
- `expandExpr(form)` — expand single form (useful for testing, Phase 3 macro compilation)
- AST API functions: `sym`, `gensym`, `array`, `append`, `isArray`, `isSymbol`, `isNumber`, `isString`, `first`, `rest`, `concat`, `length`, `nth`

## Implementation Order

```
Step 1: AST Node API functions (foundation)
Step 2: quote resolution (simplest dispatch entry)
Step 3: Sugar transforms: cons, list, car, cdr, cadr, cddr
Step 4: Quasiquote expansion — Bawden's algorithm (hardest part)
Step 5: as desugaring (simple rename + whole-and-destructure)
Step 6: Dispatch table + expansion walk (ties everything together)
Step 7: Integration — wire expand() into index.js and lykn-browser.js
```

## Step 1: AST Node API

These construct and inspect the internal AST representation. All are pure functions except `gensym` (uses counter).

```js
/** @type {number} */
let gensymCounter = 0;

/**
 * Create a symbol (atom) AST node.
 * @param {string} name
 * @returns {AtomNode}
 */
export function sym(name) {
  return { type: 'atom', value: name };
}

/**
 * Create a unique symbol for hygienic macro expansion.
 * @param {string} [prefix='g']
 * @returns {AtomNode}
 */
export function gensym(prefix = 'g') {
  return { type: 'atom', value: `${prefix}__gensym${gensymCounter++}` };
}

/**
 * Create a list AST node from items.
 * @param {...AstNode} items
 * @returns {ListNode}
 */
export function array(...items) {
  return { type: 'list', values: items };
}

/**
 * Concatenate list node values into a single list node.
 * @param {...ListNode} arrays
 * @returns {ListNode}
 * @throws {Error} If any argument is not a list node
 */
export function append(...arrays) {
  const values = [];
  for (const arr of arrays) {
    if (arr.type !== 'list') {
      throw new Error(`append: expected list node, got ${arr.type}`);
    }
    values.push(...arr.values);
  }
  return { type: 'list', values };
}

/**
 * Test whether a node is a list AST node.
 * @param {AstNode} x
 * @returns {boolean}
 */
export function isArray(x) {
  return x !== null && x !== undefined && x.type === 'list';
}

/**
 * Test whether a node is an atom (symbol) AST node.
 * @param {AstNode} x
 * @returns {boolean}
 */
export function isSymbol(x) {
  return x !== null && x !== undefined && x.type === 'atom';
}

/**
 * @param {AstNode} x
 * @returns {boolean}
 */
export function isNumber(x) {
  return x !== null && x !== undefined && x.type === 'number';
}

/**
 * @param {AstNode} x
 * @returns {boolean}
 */
export function isString(x) {
  return x !== null && x !== undefined && x.type === 'string';
}

/**
 * Get first element of a list node.
 * @param {ListNode} arr
 * @returns {AstNode | undefined}
 */
export function first(arr) {
  return arr.values[0];
}

/**
 * Get all but first element as a new list node.
 * @param {ListNode} arr
 * @returns {ListNode}
 */
export function rest(arr) {
  return { type: 'list', values: arr.values.slice(1) };
}

/** Alias for append. */
export const concat = append;

/**
 * @param {ListNode} arr
 * @returns {number}
 */
export function length(arr) {
  return arr.values.length;
}

/**
 * @param {ListNode} arr
 * @param {number} n
 * @returns {AstNode | undefined}
 */
export function nth(arr, n) {
  return arr.values[n];
}
```

## Step 2: Quote Resolution

`(quote x)` → return the form unchanged. The `"none"` walk strategy in the dispatch table handles this: no recursion into children.

When the expansion walk encounters `quote`, it returns the entire `(quote x)` form as-is. The compiler will eventually see it and throw (safety net) — but in practice, quote forms are consumed by quasiquote expansion in macro bodies, never reaching the compiler.

## Step 3: Sugar Transforms

Each receives `args` (the list values after the head atom) and returns a new AST node. The result is re-expanded by the walk.

```js
function desugarCons(args) {
  if (args.length !== 2) {
    throw new Error('cons requires exactly 2 arguments');
  }
  return array(sym('array'), args[0], args[1]);
}

function desugarList(args) {
  if (args.length === 0) {
    return sym('null');
  }
  // (list a b c) → (array a (array b (array c null)))
  let result = sym('null');
  for (let i = args.length - 1; i >= 0; i--) {
    result = array(sym('array'), args[i], result);
  }
  return result;
}

function desugarCar(args) {
  if (args.length !== 1) throw new Error('car requires exactly 1 argument');
  return array(sym('get'), args[0], { type: 'number', value: 0 });
}

function desugarCdr(args) {
  if (args.length !== 1) throw new Error('cdr requires exactly 1 argument');
  return array(sym('get'), args[0], { type: 'number', value: 1 });
}

function desugarCadr(args) {
  if (args.length !== 1) throw new Error('cadr requires exactly 1 argument');
  return array(sym('get'), array(sym('get'), args[0], { type: 'number', value: 1 }), { type: 'number', value: 0 });
}

function desugarCddr(args) {
  if (args.length !== 1) throw new Error('cddr requires exactly 1 argument');
  return array(sym('get'), array(sym('get'), args[0], { type: 'number', value: 1 }), { type: 'number', value: 1 });
}
```

### Sugar Transform Examples

| Input | Output |
|-------|--------|
| `(cons x y)` | `(array x y)` |
| `(list a b c)` | `(array a (array b (array c null)))` |
| `(list a)` | `(array a null)` |
| `(list)` | `null` |
| `(car x)` | `(get x 0)` |
| `(cdr x)` | `(get x 1)` |
| `(cadr x)` | `(get (get x 1) 0)` |
| `(cddr x)` | `(get (get x 1) 1)` |

## Step 4: Bawden's Quasiquote Algorithm

### Core function: `expandQuasiquote(form, depth)`

Called when the walk encounters `(quasiquote form)`. Receives the BODY (not the wrapping list).

**Self-evaluating atoms:**
- Numbers, strings → return as-is (they evaluate to themselves)
- Atoms at depth 0 → `(quote atom)` (quote the symbol)
- Atoms at depth > 0 → `(quote atom)` (same — symbols are always quoted)

**Lists:**
1. Check head for `quasiquote` → increment depth, recurse
2. Check head for `unquote` → if depth 0: return the argument directly. If depth > 0: decrement, recurse
3. Check head for `unquote-splicing` at top level → error (splice not in list)
4. General list → map each element through `expandQuasiquoteElement`

**Element expansion:**

Each list element produces `{ node, isSplice, isLiteral }`:

| Element | At depth 0 | Result |
|---------|-----------|--------|
| `(unquote x)` | Fires | `{ node: x, isSplice: false, isLiteral: false }` |
| `(unquote-splicing x)` | Fires | `{ node: x, isSplice: true, isLiteral: false }` |
| `(quasiquote ...)` | Increment depth | `{ node: expanded, isSplice: false, isLiteral: false }` |
| Nested list | Recurse | `{ node: expanded, isSplice: false, isLiteral: false }` |
| Atom | Quote it | `{ node: (quote atom), isSplice: false, isLiteral: true }` |
| Number/string | Self-eval | `{ node: value, isSplice: false, isLiteral: true }` |

**Optimizations:**
- All elements literal → return the form as a direct literal (no `array`/`append`)
- No splices → use `(array e1 e2 ...)` instead of `(append (array e1) (array e2) ...)`
- Has splices → use `(append ...)` with each non-splice wrapped in `(array ...)`

### DD-10 Worked Examples (optimized output)

**Example 1** — no unquotes:
```
`(if true (console:log "yes"))
→ direct literal structure (all-literal optimization)
```

**Example 2** — unquote, no splice:
```
`(if ,test (console:log "yes"))
→ (array (quote if) test (array (quote console:log) (quote "yes")))
```

**Example 3** — has splice:
```
`(if ,test ,@body)
→ (append (array (quote if)) (array test) body)
```

**Example 4** — nested list with splice:
```
`(let ((,name ,value)) ,@body)
→ (append (array (quote let)) (array (array name value)) body)
```

|-----------|---------------|

## Step 5: `as` Desugaring

Two modes:

**Simple rename** (first arg is atom):
```
(as source target) → (alias source target)
```

**Whole-and-destructure** (first arg is list/pattern):
```
(const (as (object a b) whole) expr)
→ two forms:
  (const whole expr)
  (const (object a b) whole)
```

The whole-and-destructure case is context-dependent — it needs the enclosing binding keyword and init expression. Implementation approach: add `const`/`let`/`var` to the dispatch table with an `"expand-binding"` walk that checks for `as` in binding position.

```js
function desugarAs(args) {
  if (args.length !== 2) throw new Error('as requires exactly 2 arguments');
  if (args[0].type === 'atom') {
    // Simple rename → (alias source target)
    return array(sym('alias'), args[0], args[1]);
  }
  // Whole-and-destructure handled by expand-binding walk
  throw new Error('as with pattern first argument must appear in binding position (const/let/var)');
}
```

The `expand-binding` handler in the dispatch table:
```js
"const": { walk: "expand-binding", keyword: "const" },
"let":   { walk: "expand-binding", keyword: "let" },
"var":   { walk: "expand-binding", keyword: "var" },
```

Handler logic:
```js
case 'expand-binding': {
  const args = form.values.slice(1);
  // Check if binding position contains (as pattern whole)
  if (args[0]?.type === 'list' && args[0].values.length >= 1 &&
      args[0].values[0]?.type === 'atom' && args[0].values[0]?.value === 'as') {
    const asArgs = args[0].values.slice(1);
    if (asArgs.length === 2 && asArgs[0].type === 'list') {
      const pattern = asArgs[0];
      const whole = asArgs[1];
      const initExpr = args[1];
      return [
        expandExpr(array(sym(entry.keyword), whole, initExpr)),
        expandExpr(array(sym(entry.keyword), pattern, whole)),
      ];
    }
  }
  // No as pattern — default recursive expansion
  return { type: 'list', values: form.values.map(sub => expandExpr(sub)) };
}
```

## Step 6: Dispatch Table + Expansion Walk

### Dispatch Table

```js
const dispatchTable = {
  "quote":          { walk: "none" },
  "macro":          { walk: "register-macro" },
  "cons":           { walk: "desugar", transform: desugarCons },
  "list":           { walk: "desugar", transform: desugarList },
  "car":            { walk: "desugar", transform: desugarCar },
  "cdr":            { walk: "desugar", transform: desugarCdr },
  "cadr":           { walk: "desugar", transform: desugarCadr },
  "cddr":           { walk: "desugar", transform: desugarCddr },
  "as":             { walk: "desugar", transform: desugarAs },
  "const":          { walk: "expand-binding", keyword: "const" },
  "let":            { walk: "expand-binding", keyword: "let" },
  "var":            { walk: "expand-binding", keyword: "var" },
  "macroexpand":    { walk: "debug-expand", mode: "full" },
  "macroexpand-1":  { walk: "debug-expand", mode: "once" },
};
```

### Walk Strategies

| Strategy | Behavior |
|----------|----------|
| `"none"` | Return form unchanged. No recursion. |
| `"register-macro"` | Error: "unexpected macro definition in expansion pass" |
| `"desugar"` | Call transform(args), re-expand result |
| `"expand-binding"` | Check for `as` in binding position → split; otherwise default recursion |
| `"debug-expand"` | Error: "not yet implemented" (Phase 3) |
| Default (not in table) | Recursively expand all sub-forms |

### Expansion Walk Algorithm

```js
const MAX_EXPAND_ITERATIONS = 1000;

export function expandExpr(form) {
  if (form.type === 'atom' || form.type === 'number' || form.type === 'string') {
    return form;
  }

  if (form.type === 'cons') {
    return { type: 'cons', car: expandExpr(form.car), cdr: expandExpr(form.cdr) };
  }

  if (form.values.length === 0) return form;

  const head = form.values[0];

  // Quasiquote
  if (head.type === 'atom' && head.value === 'quasiquote') {
    if (form.values.length !== 2) throw new Error('quasiquote requires exactly one argument');
    return expandExpr(expandQuasiquote(form.values[1], 0));
  }

  // Unquote/splice outside quasiquote
  if (head.type === 'atom' && head.value === 'unquote') {
    throw new Error('unquote outside of quasiquote');
  }
  if (head.type === 'atom' && head.value === 'unquote-splicing') {
    throw new Error('unquote-splicing outside of quasiquote');
  }

  // Dispatch table
  if (head.type === 'atom') {
    const entry = dispatchTable[head.value];
    if (entry) {
      // handle per walk strategy (see above)
    }
  }

  // Default: expand all sub-forms
  return { type: 'list', values: form.values.map(sub => expandExpr(sub)) };
}

export function expand(forms) {
  const result = [];
  for (const form of forms) {
    const expanded = expandExpr(form);
    if (Array.isArray(expanded)) {
      result.push(...expanded);
    } else {
      result.push(expanded);
    }
  }
  return result;
}
```
